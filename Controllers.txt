public class SocioController {
	//objetos vista y modelo
	private SocioView vista;
	private Socio modelo;
 
	//constructor para inicializar el modelo y la vista
	public SocioController(Socio modelo, SocioView vista) {
		this.modelo = modelo;
		this.vista = vista;
	}
 
	//getters y setters para el modelo
	public int getCUIT() {
		return modelo.getCUIT();
	}
	public void setCUIT(int cuit) {
		this.modelo.setCUIT(cuit);
	}
	public String getEstado() {
		return modelo.getEstado();
	}
	public void setEstado(String estado) {
		this.modelo.setEstado(estado); 
	}
	public String getTipoSocio() {
		return modelo.getTipoSocio();
	}
	public void setTipoSocio(String tiposocio) {
		this.modelo.setTipoSocio(tiposocio);
	}
	
	//pasa el modelo a la vista para presentar los datos
	public void actualizarVista() {
		vista.imprimirDatosSocio(modelo.getCUIT(),modelo.getEstado(), modelo.getTiposocio());
	}
